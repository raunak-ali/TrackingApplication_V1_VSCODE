using System;  
  public class SwapExample  
   {  
     public static void Main()  
      {  
       int  a=5, b=10;            
       a=a*b; //a=50 (5*10)      
       b=a/b; //b=5 (50/10)      
       a=a/b; //a=10 (50/5)    
       Console.Write("int: 5 float: 1.05 double: 6.89989 long: 1234567890 string: Hello");       
}
}
